Функция Задание1() 
	

	
	СуммаВсего = 100;
	
	ТаблицаРаспределение = Новый ТаблицаЗначений;

	ТаблицаРаспределение.Колонки.Добавить("Количество", Новый ОписаниеТипов("Число", Новый КвалификаторыЧисла(10,0)));

	ТаблицаРаспределение.Колонки.Добавить("Сумма", Новый ОписаниеТипов("Число", Новый КвалификаторыЧисла(15,2)));

	Строка = ТаблицаРаспределение.Добавить();

	Строка.Количество = 2;

	Строка = ТаблицаРаспределение.Добавить();

	Строка.Количество = 3;

	Строка = ТаблицаРаспределение.Добавить();

	Строка.Количество = 4;   
	
	ВсегоКоличество = 0;

    	Для каждого Строка Из ТаблицаРаспределение Цикл
	
		ВсегоКоличество = ВсегоКоличество + Строка.Количество;	
	
	КонецЦикла; 
	
	СуммаБезПогрешности = 0;
	
	Для каждого Строка Из ТаблицаРаспределение Цикл
		
		А = Строка.количество / ВсегоКоличество * СуммаВсего;
		Строка.Сумма = А;
		СуммаБезПогрешности = СуммаБезПогрешности +  А;
	
	КонецЦикла;
	
	Если ТаблицаРаспределение.Итог("Сумма") <> СуммаВсего Тогда
	
		
		СуммаВсего = СуммаВсего - (СуммаБезПогрешности - ТаблицаРаспределение.Итог("Сумма")); 	
		//В этой строке я при возможном появлении неточности из-за округления 
		//уменьшаю исходную сумму на возможную погрешность 		
		//Понятно, что если СуммаВсего=СуммаБезПогрешности , то задача решена правильно
	КонецЕсли; 
	
	Возврат(ТаблицаРаспределение.Итог("Сумма") = СуммаВсего);

КонецФункции
